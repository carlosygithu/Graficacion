```python
# Importa OpenCV para procesamiento de imágenes
import cv2 as cv

# Importa NumPy para crear matrices
import numpy as np


# Crea una imagen de 300x300 píxeles
# Todos los valores son 0 (negro)
img = np.zeros((300,300), dtype=np.uint8)


# Crea un cuadrado dentro de la imagen
# El valor 120 representa gris
img[50:250,50:250] = 120


# OPERADOR NEGATIVO
# Invierte los colores
# Negro se convierte en blanco
# Blanco se convierte en negro
negativo = 255 - img


# OPERADOR AUMENTAR BRILLO
# Suma 50 a cada píxel
# La imagen se vuelve más clara
brillo = img + 50


# OPERADOR DISMINUIR BRILLO
# Resta 50 a cada píxel
# La imagen se vuelve más oscura
oscuro = img - 50


# OPERADOR UMBRAL
# Convierte la imagen a blanco y negro puro
# Valores mayores a 100 se convierten en 255
# Valores menores se convierten en 0
_, umbral = cv.threshold(img, 100, 255, cv.THRESH_BINARY)


# OPERADOR INVERSIÓN MANUAL
# Crea una nueva imagen vacía
invertida = np.zeros_like(img)


# Recorre cada píxel
for i in range(img.shape[0]):
    
    for j in range(img.shape[1]):
        
        # Invierte el valor del píxel
        invertida[i,j] = 255 - img[i,j]


# Muestra la imagen original
cv.imshow("Original", img)

# Muestra la imagen negativa
cv.imshow("Negativo", negativo)

# Muestra imagen con más brillo
cv.imshow("Brillo aumentado", brillo)

# Muestra imagen con menos brillo
cv.imshow("Brillo disminuido", oscuro)

# Muestra imagen con umbral
cv.imshow("Umbral", umbral)

# Muestra imagen invertida manualmente
cv.imshow("Invertida manual", invertida)


# Espera tecla del usuario
cv.waitKey(0)


# Cierra ventanas
cv.destroyAllWindows()
