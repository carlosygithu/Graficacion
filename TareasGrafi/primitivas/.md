## Código 

```python
# Importa OpenCV para procesamiento de imágenes
import cv2 as cv

# Importa NumPy para crear matrices
import numpy as np 

# Crea una imagen de 500x500 píxeles
# np.ones crea una matriz llena de 1
# Se multiplica por 150 para crear un fondo gris
# dtype uint8 permite valores de 0 a 255
img = np.ones((500,500), np.uint8)*150 

# Dibuja un rectángulo sobre la imagen
# (10,10) es la esquina superior izquierda
# (200,100) es la esquina inferior derecha
# (34,56,100) es el color
# -1 significa que el rectángulo estará relleno
cv.rectangle(img, (10,10), (200,100), (34,56,100), -1)

# Dibuja un círculo pequeño
# (255,255) es el centro del círculo
# 1 es el radio
# (23,43,144) es el color
# -1 significa círculo relleno
cv.circle(img, (255,255), 1, (23, 43, 144), -1 )

# Dibuja una línea sobre la imagen
# (255,255) es el punto inicial
# (200,100) es el punto final
# (23,244,144) es el color
# 4 es el grosor de la línea
cv.line(img, (255,255), (200,100), (23, 244, 144), 4)


# Ciclo que se repite 400 veces
# Se usa para crear animación
for i in range(400):

    # Dibuja un círculo cuyo centro cambia
    # El centro es (i,i)
    # El radio también aumenta
    # Esto crea un efecto de crecimiento
    cv.circle(img, (i,i), i , (255, 0, 0), -1 )

    # Dibuja un rectángulo que se mueve
    # La posición cambia con el valor de i
    cv.rectangle(img, (10+i,10), (200,100), (34,56,100), -1)

    # Muestra la imagen en cada iteración
    # Esto permite ver la animación
    cv.imshow('img', img)

    # Espera 30 milisegundos
    # Controla la velocidad de la animación
    cv.waitKey(30)
    

# Muestra la imagen final
cv.imshow('img', img)

# Espera hasta que el usuario presione una tecla
cv.waitKey(0)

# Cierra todas las ventanas abiertas
cv.destroyAllWindows()
